56 void Pi3HatInterface::cycle(
57  std::function<void(const moteus::Output&)> callback) {
58  const moteus::Data& data = this->data_;
59  std::lock_guard<std::mutex> lock(mutex_);
60  if (ongoing_can_cycle_) {
61  throw std::logic_error(
62  "Cycle cannot be called before the previous one has completed.");
63  }
64 
65  callback_ = std::move(callback);
66  ongoing_can_cycle_ = true;
67  data_ = data;
68 
69  can_wait_condition_.notify_all();
70 }

72 void Pi3HatInterface::run_can_thread() {
73  vulp::utils::configure_cpu(can_cpu_);
74  vulp::utils::configure_scheduler(10);
75  pi3hat_.reset(new Pi3Hat({pi3hat_config_}));
76  pthread_setname_np(pthread_self(), "can_thread");
77  while (!done_) {
78  {
79  std::unique_lock<std::mutex> lock(mutex_);
80  if (!ongoing_can_cycle_) {
81  can_wait_condition_.wait(lock);
82  if (done_) {
83  return;
84  }
85  if (!ongoing_can_cycle_) {
86  continue;
87  }
88  }
89  }
90  auto output = cycle_can_thread();
91  std::function<void(const moteus::Output&)> callback_copy;
92  {
93  std::unique_lock<std::mutex> lock(mutex_);
94  ongoing_can_cycle_ = false;
95  std::swap(callback_copy, callback_);
96  }
97  callback_copy(output);
98  }
99 }

101 moteus::Output Pi3HatInterface::cycle_can_thread() {
102  tx_can_.resize(data_.commands.size());
103  int out_idx = 0;
104  for (const auto& cmd : data_.commands) {
105  const auto& query = cmd.query;
106 
107  auto& can = tx_can_[out_idx++];
108 
109  can.expect_reply = query.any_set();
110  can.id = cmd.id | (can.expect_reply ? 0x8000 : 0x0000);
111  can.size = 0;
112 
113  can.bus = [&]() {
114  const auto it = servo_bus_map().find(cmd.id);
115  if (it == servo_bus_map().end()) {
116  return 1;
117  }
118  return it->second;
119  }();
120 
121  moteus::WriteCanFrame write_frame(can.data, &can.size);
122  switch (cmd.mode) {
123  case moteus::Mode::kStopped: {
124  moteus::EmitStopCommand(&write_frame);
125  break;
126  }
127  case moteus::Mode::kPosition:
128  case moteus::Mode::kZeroVelocity: {
129  moteus::EmitPositionCommand(&write_frame, cmd.position, cmd.resolution);
130  break;
131  }
132  default: {
133  throw std::logic_error("unsupported mode");
134  }
135  }
136  moteus::EmitQueryCommand(&write_frame, cmd.query);
137  }
138 
139  rx_can_.resize(data_.commands.size() * 2);
140 
141  Pi3Hat::Input input;
142  input.tx_can = {tx_can_.data(), tx_can_.size()};
143  input.rx_can = {rx_can_.data(), rx_can_.size()};
144  input.attitude = &attitude_;
145  input.request_attitude = true;
146  input.wait_for_attitude = true; // we may revisit this later
147 
148  moteus::Output result;
149  const auto pi3hat_output = pi3hat_->Cycle(input);
150  if (pi3hat_output.error) {
151  spdlog::error("pi3hat: {}", pi3hat_output.error);
152  }
153  for (size_t i = 0; i < pi3hat_output.rx_can_size && i < data_.replies.size();
154  ++i) {
155  const auto& can = rx_can_[i];
156  data_.replies[i].id = (can.id & 0x7f00) >> 8;
157  data_.replies[i].result = moteus::ParseQueryResult(can.data, can.size);
158  result.query_result_size = i + 1;
159  }
160  if (!pi3hat_output.attitude_present) { // because we wait for attitude
161  spdlog::warn("Missing attitude data!");
162  }
163  return result;
164 }
