82 void Spine::run() {
83  Dictionary& spine = working_dict_("spine");
84  utils::SynchronousClock clock(frequency_);
85  while (state_machine_.state() != State::kOver) {
86  cycle();
87  if (state_machine_.state() != State::kSendStops) {
88  spine("clock")("measured_period") = clock.measured_period();
89  spine("clock")("skip_count") = clock.skip_count();
90  spine("clock")("slack") = clock.slack();
91  log_working_dict();
92  }
93  clock.wait_for_next_tick();
94  }
95  spdlog::info("SEE YOU SPACE COWBOY...");
96 }

104 void Spine::simulate(unsigned nb_substeps) {
105  while (state_machine_.state() != State::kOver) {
106  begin_cycle();
107  if (state_machine_.state() == State::kReset) {
108  cycle_actuation(); // S1: cycle the simulator, promise actuation_output_
109  cycle_actuation(); // S2: fill latest_replies_ from actuation_output_
110  cycle_actuation(); // S3: fill observation dict from latest_replies_
111  // now the first observation is ready to be read by the agent
112  } else if (state_machine_.state() == State::kAct) {
113  for (unsigned substep = 0; substep < nb_substeps; ++substep) {
114  cycle_actuation();
115  }
116  }
117  end_cycle();
118  }
119 }

121 void Spine::begin_cycle() {
122  if (caught_interrupt_) {
123  state_machine_.process_event(Event::kInterrupt);
124  } else /* (!caught_interrupt_) */ {
125  state_machine_.process_event(Event::kCycleBeginning);
126  }
127  state_cycle_beginning_ = state_machine_.state();
128 
129  // Read input dictionary if applicable
130  if (state_machine_.state() == State::kReset) {
131  Dictionary& config = working_dict_("config");
132  const char* data = agent_interface_.data();
133  size_t size = agent_interface_.size();
134  config.clear();
135  config.update(data, size);
136  reset(config);
137  } else if (state_machine_.state() == State::kAct) {
138  Dictionary& action = working_dict_("action");
139  const char* data = agent_interface_.data();
140  size_t size = agent_interface_.size();
141  action.update(data, size);
142  }
143 }

145 void Spine::end_cycle() {
146  // Write observation if applicable
147  const Dictionary& observation = working_dict_("observation");
148  working_dict_("time") = observation.get<double>("time");
149  if (state_machine_.state() == State::kObserve) {
150  size_t size = observation.serialize(ipc_buffer_);
151  agent_interface_.write(ipc_buffer_.data(), size);
152  }
153 
154  state_machine_.process_event(Event::kCycleEnd);
155  state_cycle_end_ = state_machine_.state();
156 }

158 void Spine::cycle_actuation() {
159  try {
160  // 1. Observation
161  Dictionary& observation = working_dict_("observation");
162  observation::observe_time(observation);
163  observation::observe_servos(observation, actuation_.servo_joint_map(),
164  latest_replies_);
165  actuation_.observe(observation);
166  // Observers need configuration, so they cannot run at stop
167  if (state_machine_.state() != State::kSendStops &&
168  state_machine_.state() != State::kShutdown) {
169  try {
170  observer_pipeline_.run(observation);
171  } catch (const ObserverError& e) {
172  spdlog::info("Key error from {}: key \"{}\" not found", e.prefix(),
173  e.key());
174  }
175  }
176 
177  // 2. Action
178  if (state_machine_.state() == State::kSendStops ||
179  state_machine_.state() == State::kShutdown) {
180  actuation_.write_stop_commands();
181  } else if (state_machine_.state() == State::kAct) {
182  const Dictionary& action = working_dict_("action");
183  actuation_.process_action(action);
184  actuation_.write_position_commands(action);
185  }
186  } catch (const std::exception& e) {
187  spdlog::error("[Spine] Caught an exception: {}", e.what());
188  spdlog::error("[Spine] Sending stop commands...");
189  state_machine_.process_event(Event::kInterrupt);
190  actuation_.write_stop_commands();
191  } catch (...) {
192  spdlog::error("[Spine] Caught an unknown exception!");
193  spdlog::error("[Spine] Sending stop commands...");
194  state_machine_.process_event(Event::kInterrupt);
195  actuation_.write_stop_commands();
196  }
197 
198  // Whatever exceptions were thrown around, we caught them and at this
199  // point every actuation command is either a stop or a position one.
200 
201  // 3. Wait for the result of the last query and copy it
202  if (actuation_output_.valid()) {
203  const auto current_values = actuation_output_.get(); // may wait here
204  const auto rx_count = current_values.query_result_size;
205  latest_replies_.resize(rx_count);
206  std::copy(actuation_.replies().begin(),
207  actuation_.replies().begin() + rx_count, latest_replies_.begin());
208  }
209 
210  // Now we are after the previous cycle (we called actuation_output_.get())
211  // and before the next one. This is a good time to break out of loop of
212  // communication cycles. Otherwise, the interface may warn that it is waiting
213  // for the last actuation cycle to finish.
214  if (state_machine_.is_over_after_this_cycle()) {
215  spdlog::info("Wrapping up last communication cycle");
216  return;
217  }
218 
219  // 4. Start a new cycle. Results have been copied, so actuation commands and
220  // replies are available again to the actuation thread for writing.
221  auto promise = std::make_shared<std::promise<actuation::moteus::Output>>();
222  actuation_.cycle([promise](const actuation::moteus::Output& output) {
223  // This is called from an arbitrary thread, so we
224  // just set the promise value here.
225  promise->set_value(output);
226  });
227  actuation_output_ = promise->get_future();
228 }
