41 void MockInterface::cycle(std::function<void(const moteus::Output&)> callback) {
42  const moteus::Data& data = this->data_;
43  assert(data.replies.size() == data.commands.size());
44 
45  for (const auto& command : data.commands) {
46  const auto servo_id = command.id;
47  auto& result = query_results_[servo_id];
48  result.mode = command.mode;
49  const auto& target = command.position;
50  if (std::isnan(target.position) && !std::isnan(target.velocity) &&
51  !std::isnan(result.position)) {
52  result.position += target.velocity * dt_;
53  } else {
54  result.position = target.position;
55  }
56  result.velocity = target.velocity;
57  result.torque = target.feedforward_torque;
58  }
59 
60  moteus::Output output;
61  for (size_t i = 0; i < data.replies.size(); ++i) {
62  const auto servo_id = data.commands[i].id;
63  data.replies[i].id = servo_id;
64  data.replies[i].result = query_results_[servo_id];
65  output.query_result_size = i + 1;
66  }
67  callback(output);
68 }
